\be
u^* = u^{n} + \tau L(u^{n})
\nd
\be
u^{**} = u^* + \tau L(u^{*})
\nd
\be
u^{n+1,*} = \frac 12 (u^{**} + u^n )
\nd
approximation:
\be
u^{n+1} = u^n + \tau L(u^{n}) + \Order(\tau^2) = u^{*} + \Order(\tau^2)
\nd
substituting:
\be
u^{n+1,*} = \frac 12 ( u^* + \tau L(u^{*}) + u^n) =
\frac 12 ( u^n + \tau L(u^n) + \tau L( u^{n+1} + \Order(\tau^2)) + u^n) 
\nd
\be
u^{n+1,*} = 
u^n + \frac 12 (  \tau L(u^n) + \tau L( u^{n+1} + \Order(\tau^2)) ) 
\nd
Hence
\be
u^{n+1,*} = 
u^n + \frac 12 (  \tau L(u^n) + \tau L( u^{n+1}) )   + \Order(\tau^3)
\nd
which is the the trapezoidal rule.
